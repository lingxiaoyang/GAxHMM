1function c = setdiff_simple(a,b)
 2% ONLY SUPPORT TWO IN ARGS and ONE OUT ARG only FOR ROW VEC
 3%SETDIFF Set difference.
 4%   C = SETDIFF(A,B) for vectors A and B, returns the values in A that 
 5%   are not in B with no repetitions. C will be sorted.
 6
 7    % Call ISMEMBER to determine list of non-matching elements of A.
 8    logUA = ~(ismember(a,b));
 9    c = a(logUA);
10    
11    % Call UNIQUE to remove duplicates from list of non-matches.
12    c = unique(c);
13end
